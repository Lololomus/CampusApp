// === ГЛАВНЫЙ АПП ===
function App() {
  const [navTab, setNavTab] = useState('dating'); // Дефолтный таб для теста
  const [subTab, setSubTab] = useState('left');
  const [datingSubTab, setDatingSubTab] = useState('left'); // Отдельный стейт для хедера дейтинга
  
  // Состояние глобального меню Создания
  const [showCreateMenu, setShowCreateMenu] = useState(false);
  
  const [isScrolled, setIsScrolled] = useState(false);
  const [isSearchExpanded, setIsSearchExpanded] = useState(false);
  const scrollRef = useRef(null);

  useEffect(() => {
    if (scrollRef.current) scrollRef.current.scrollTop = 0;
    setIsScrolled(false);
    setIsSearchExpanded(false);
    setSubTab('left');
  }, [navTab]);

  useEffect(() => {
    const handleScroll = (e) => {
      const y = e.target.scrollTop;
      if (y > 40 && !isScrolled) setIsScrolled(true);
      else if (y <= 40 && isScrolled) {
        setIsScrolled(false);
        setIsSearchExpanded(false);
      }
    };
    const node = scrollRef.current;
    if (node) node.addEventListener('scroll', handleScroll);
    return () => node && node.removeEventListener('scroll', handleScroll);
  }, [isScrolled]);


  // Конфигурация табов
  const getTabConfig = () => {
    switch(navTab) {
      case 'feed': return { title: 'Campus.', hasSub: true, subLeft: 'Посты', subRight: 'Запросы', hasSearch: true, accent: theme.primary };
      case 'market': return { title: 'Барахолка', hasSub: true, subLeft: 'Товары', subRight: 'Избранное', hasSearch: true, accent: theme.primary };
      case 'dating': return { title: 'Знакомства', hasSub: false, hasSearch: false, isCustomDatingHeader: true }; // Кастомная шапка
      case 'profile': return { title: 'Профиль', hasSub: false, hasSearch: false };
      default: return {};
    }
  };

  const config = getTabConfig();
  const showSearchDrawer = config.hasSearch && (!isScrolled || isSearchExpanded);
  
  const getPaddingTop = () => {
     if (config.isCustomDatingHeader) return '110px';
     if (config.hasSub && config.hasSearch) return '230px';
     if (config.hasSub) return '160px';
     return '110px';
  };

  return (
    <div style={{ position: 'relative', width: '390px', height: '844px', backgroundColor: '#000', borderRadius: '46px', border: '10px solid #222', overflow: 'hidden', display: 'flex', flexDirection: 'column' }}>
      
      {/* Notch */}
      <div style={{ position: 'absolute', top: 8, left: '50%', transform: 'translateX(-50%)', width: '120px', height: '32px', backgroundColor: '#000', borderRadius: '16px', zIndex: 1100 }} />

      {/* Освещение шапки */}
      <div style={{
        position: 'absolute', top: 0, left: 0, right: 0, height: '140px',
        background: navTab === 'dating' ? 'rgba(255, 45, 85, 0.05)' : navTab === 'market' ? 'rgba(0, 230, 118, 0.03)' : 'rgba(212, 255, 0, 0.03)',
        backdropFilter: 'blur(12px)', WebkitBackdropFilter: 'blur(12px)',
        maskImage: 'linear-gradient(to bottom, black 40%, transparent 100%)',
        zIndex: 51, pointerEvents: 'none', transition: 'background 0.5s'
      }} />

      {/* ШАПКА */}
      <header style={{ position: 'absolute', top: 56, left: 16, right: 16, zIndex: 100, display: 'flex', flexDirection: 'column' }}>
        
        {/* Заголовок для обычных табов */}
        {!config.isCustomDatingHeader && (
          <div style={{ height: 28, marginBottom: 12, display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
            <h1 style={{ margin: 0, fontSize: navTab === 'feed' ? '24px' : '18px', fontWeight: 800, color: '#FFF', letterSpacing: '-0.5px' }}>
              {config.title}
            </h1>
          </div>
        )}

        {/* Кастомный хедер для Дейтинга (Анкеты - Аватар - Симпатии) */}
        {config.isCustomDatingHeader && (
          <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between', height: '44px', padding: '0 8px' }}>
            <button className="spring-btn" onClick={() => setDatingSubTab('left')} style={{ flex: 1, background: datingSubTab === 'left' ? 'rgba(255, 45, 85, 0.15)' : 'transparent', color: datingSubTab === 'left' ? theme.datingAccent : '#888', padding: '8px 0', borderRadius: '20px', fontWeight: 700, fontSize: '15px', border: 'none', transition: 'all 0.2s' }}>Анкеты</button>
            
            <div className="spring-btn" onClick={() => setDatingSubTab('center')} style={{ width: 44, height: 44, borderRadius: '22px', border: datingSubTab === 'center' ? `2px solid ${theme.datingAccent}` : '2px solid transparent', padding: 2, flexShrink: 0, margin: '0 12px', cursor: 'pointer', transition: 'all 0.2s' }}>
              <div style={{ width: '100%', height: '100%', borderRadius: '50%', background: 'linear-gradient(135deg, #7A5CFF, #4DA6FF)', display: 'flex', alignItems: 'center', justifyContent: 'center', fontSize: '18px', fontWeight: 800, color: '#FFF' }}>Ы</div>
            </div>

            <button className="spring-btn" onClick={() => setDatingSubTab('right')} style={{ flex: 1, background: datingSubTab === 'right' ? 'rgba(255, 45, 85, 0.15)' : 'transparent', color: datingSubTab === 'right' ? theme.datingAccent : '#888', padding: '8px 0', borderRadius: '20px', fontWeight: 700, fontSize: '15px', border: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', gap: '6px', transition: 'all 0.2s' }}>
              Симпатии <span style={{ background: theme.text, color: '#000', borderRadius: '10px', padding: '2px 6px', fontSize: '12px' }}>6</span>
            </button>
          </div>
        )}

        {/* Саб-табы обычные */}
        {config.hasSub && !config.isCustomDatingHeader && (
          <div style={{ display: 'flex', alignItems: 'center', height: '44px', width: '100%' }}>
            <div style={{ flex: 1, height: '100%', background: theme.surfaceElevated, borderRadius: '22px', display: 'flex', padding: '4px' }}>
              <div style={{ position: 'relative', width: '100%', display: 'flex' }}>
                 <div style={{
                    position: 'absolute', top: 0, bottom: 0, width: 'calc(50% - 4px)',
                    background: config.accent, borderRadius: '18px',
                    transform: `translateX(${subTab === 'left' ? '0' : '100%'})`,
                    transition: 'transform 0.3s cubic-bezier(0.32, 0.72, 0, 1)', zIndex: 1,
                    boxShadow: `0 2px 10px ${config.accent}40`
                  }} />
                  <button onClick={() => setSubTab('left')} style={{ flex: 1, position: 'relative', zIndex: 2, background: 'none', border: 'none', color: subTab === 'left' ? '#000' : '#FFF', fontWeight: 700, fontSize: '14px', cursor: 'pointer', transition: 'color 0.3s' }}>{config.subLeft}</button>
                  <button onClick={() => setSubTab('right')} style={{ flex: 1, position: 'relative', zIndex: 2, background: 'none', border: 'none', color: subTab === 'right' ? '#000' : '#FFF', fontWeight: 700, fontSize: '14px', cursor: 'pointer', transition: 'color 0.3s' }}>{config.subRight}</button>
              </div>
            </div>

            {config.hasSearch && (
              <div style={{ width: isScrolled ? '44px' : '0px', marginLeft: isScrolled ? '8px' : '0px', opacity: isScrolled ? 1 : 0, overflow: 'hidden', transition: 'all 0.3s', display: 'flex', justifyContent: 'center' }}>
                <button onClick={() => setIsSearchExpanded(!isSearchExpanded)} className="spring-btn" style={{ width: '44px', height: '44px', borderRadius: '22px', background: isSearchExpanded ? config.accent : theme.surfaceElevated, border: 'none', color: isSearchExpanded ? '#000' : '#FFF', display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'pointer', flexShrink: 0 }}>
                  {isSearchExpanded ? <XIcon size={20} /> : <Search size={20} />}
                </button>
              </div>
            )}
          </div>
        )}

        {/* Выезжающий поиск */}
        {config.hasSearch && (
          <div style={{ display: 'grid', gridTemplateRows: showSearchDrawer ? '1fr' : '0fr', transition: 'grid-template-rows 0.3s cubic-bezier(0.32, 0.72, 0, 1)' }}>
            <div style={{ minHeight: 0, overflow: 'hidden' }}>
              <div style={{ paddingTop: '12px', paddingBottom: '4px', display: 'flex', flexDirection: 'column', gap: '12px', opacity: showSearchDrawer ? 1 : 0, transform: showSearchDrawer ? 'translateY(0)' : 'translateY(-10px)', transition: 'all 0.3s' }}>
                <div style={{ display: 'flex', alignItems: 'center', height: '44px', padding: '0 16px', background: '#161618', borderRadius: '22px', border: `1px solid ${theme.border}` }}>
                  <Search size={18} style={{ color: theme.textMuted, marginRight: '10px' }} />
                  <input placeholder={`Поиск в ${config.title.toLowerCase()}...`} style={{ flex: 1, height: '100%', background: 'transparent', border: 'none', color: '#FFF', fontSize: '15px', outline: 'none' }} />
                </div>
              </div>
            </div>
          </div>
        )}
      </header>

      {/* КОНТЕНТ */}
      <div ref={scrollRef} className="hide-scroll" style={{ flex: 1, overflowY: 'auto', paddingTop: getPaddingTop(), paddingBottom: '120px', zIndex: 10, transition: 'padding-top 0.3s' }}>
        {navTab === 'feed' && <div style={{textAlign:'center', color: '#888', marginTop: 100}}>Лента (Реализована ранее)</div>}
        {navTab === 'market' && <MarketView />}
        {navTab === 'dating' && <DatingView />}
        {navTab === 'profile' && <ProfileView />}
        <div style={{ height: '40px' }}/>
      </div>

      {/* НИЖНИЙ ТАББАР */}
      <div style={{ position: 'absolute', bottom: '16px', left: '16px', right: '16px', zIndex: 100, display: 'flex', justifyContent: 'center' }}>
        <nav style={{ background: theme.surfaceElevated, borderRadius: '32px', padding: '8px 12px', display: 'flex', alignItems: 'center', justifyContent: 'space-between', width: '100%', boxShadow: '0 20px 40px rgba(0,0,0,0.8), inset 0 1px 0 rgba(255,255,255,0.05)' }}>
          {[ 
            { id: 'feed', icon: Home }, 
            { id: 'market', icon: ShoppingBag }, 
            { id: 'create', isAction: true }, 
            { id: 'dating', icon: Heart }, 
            { id: 'profile', icon: User } 
          ].map((tab, i) => {
            if (tab.isAction) return (
              <button 
                key="create" 
                onClick={() => setShowCreateMenu(true)}
                className="spring-btn" 
                style={{ width: '48px', height: '48px', borderRadius: '24px', background: theme.primary, border: 'none', color: '#000', display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'pointer', margin: '0 8px', flexShrink: 0, boxShadow: `0 4px 16px ${theme.primary}40`, transition: 'transform 0.2s' }}
                style={showCreateMenu ? { transform: 'rotate(45deg)', background: '#222', color: '#FFF', width: '48px', height: '48px', borderRadius: '24px', border: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'pointer', margin: '0 8px', flexShrink: 0 } : { width: '48px', height: '48px', borderRadius: '24px', background: theme.primary, border: 'none', color: '#000', display: 'flex', alignItems: 'center', justifyContent: 'center', cursor: 'pointer', margin: '0 8px', flexShrink: 0, boxShadow: `0 4px 16px ${theme.primary}40`, transition: 'transform 0.2s, background 0.2s' }}
              >
                <Plus size={26} strokeWidth={3} style={{ transition: 'transform 0.2s' }} />
              </button>
            );
            const Icon = tab.icon;
            const isActive = navTab === tab.id;
            return (
              <button key={tab.id} onClick={() => setNavTab(tab.id)} className="spring-btn" style={{ flex: 1, height: '48px', background: 'transparent', border: 'none', display: 'flex', flexDirection: 'column', alignItems: 'center', justifyContent: 'center', color: isActive ? '#FFFFFF' : theme.textMuted, cursor: 'pointer', position: 'relative' }}>
                {isActive && <div style={{ position: 'absolute', top: -8, width: '24px', height: '4px', borderRadius: '0 0 4px 4px', background: '#FFFFFF', boxShadow: `0 4px 12px rgba(255,255,255,0.4)` }}/>}
                <Icon size={22} strokeWidth={isActive ? 2.5 : 2} />
              </button>
            );
          })}
        </nav>
      </div>

      {/* ШТОРКА СОЗДАНИЯ КОНТЕНТА */}
      <div 
        onClick={() => setShowCreateMenu(false)}
        style={{
          position: 'absolute', top: 0, left: 0, right: 0, bottom: 0,
          background: 'rgba(0,0,0,0.6)', backdropFilter: 'blur(4px)', WebkitBackdropFilter: 'blur(4px)',
          opacity: showCreateMenu ? 1 : 0, pointerEvents: showCreateMenu ? 'auto' : 'none',
          zIndex: 200, transition: 'opacity 0.3s', display: 'flex', flexDirection: 'column', justifyContent: 'flex-end'
        }}
      >
        <div 
          onClick={e => e.stopPropagation()}
          style={{
            background: theme.surfaceElevated, borderTopLeftRadius: '24px', borderTopRightRadius: '24px',
            padding: '24px 16px 40px', transform: showCreateMenu ? 'translateY(0)' : 'translateY(100%)',
            transition: 'transform 0.3s cubic-bezier(0.32, 0.72, 0, 1)', borderTop: `1px solid ${theme.border}`
          }}
        >
          <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: '20px' }}>
            <h3 style={{ margin: 0, fontSize: '20px', fontWeight: 800, color: '#FFF' }}>Что создать?</h3>
            <button onClick={() => setShowCreateMenu(false)} style={{ background: 'none', border: 'none', color: theme.textMuted, cursor: 'pointer' }}><XIcon size={24} /></button>
          </div>
          
          <div style={{ display: 'flex', flexDirection: 'column', gap: '12px' }}>
            <button className="spring-btn" style={{ width: '100%', background: '#2C2C2E', border: 'none', borderRadius: '16px', padding: '16px', display: 'flex', alignItems: 'center', gap: '16px', color: '#FFF' }}>
               <div style={{ width: 40, height: 40, borderRadius: '12px', background: 'rgba(212,255,0,0.1)', color: theme.primary, display: 'flex', alignItems: 'center', justifyContent: 'center' }}><PenTool size={20} /></div>
               <div style={{ textAlign: 'left' }}>
                 <div style={{ fontWeight: 700, fontSize: '16px' }}>Новый пост</div>
                 <div style={{ color: theme.textMuted, fontSize: '13px', marginTop: '2px' }}>Поделиться мыслями в ленте</div>
               </div>
            </button>
            
            <button className="spring-btn" style={{ width: '100%', background: '#2C2C2E', border: 'none', borderRadius: '16px', padding: '16px', display: 'flex', alignItems: 'center', gap: '16px', color: '#FFF' }}>
               <div style={{ width: 40, height: 40, borderRadius: '12px', background: 'rgba(77, 166, 255, 0.1)', color: '#4DA6FF', display: 'flex', alignItems: 'center', justifyContent: 'center' }}><MessageSquare size={20} /></div>
               <div style={{ textAlign: 'left' }}>
                 <div style={{ fontWeight: 700, fontSize: '16px' }}>Запрос помощи</div>
                 <div style={{ color: theme.textMuted, fontSize: '13px', marginTop: '2px' }}>Задать вопрос студентам</div>
               </div>
            </button>

            <button className="spring-btn" style={{ width: '100%', background: '#2C2C2E', border: 'none', borderRadius: '16px', padding: '16px', display: 'flex', alignItems: 'center', gap: '16px', color: '#FFF' }}>
               <div style={{ width: 40, height: 40, borderRadius: '12px', background: 'rgba(0, 230, 118, 0.1)', color: theme.marketAccent, display: 'flex', alignItems: 'center', justifyContent: 'center' }}><Tag size={20} /></div>
               <div style={{ textAlign: 'left' }}>
                 <div style={{ fontWeight: 700, fontSize: '16px' }}>Объявление на маркет</div>
                 <div style={{ color: theme.textMuted, fontSize: '13px', marginTop: '2px' }}>Продать учебники или вещи</div>
               </div>
            </button>
          </div>
        </div>
      </div>

    </div>
  );
}
