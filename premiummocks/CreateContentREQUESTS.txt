      function SmartDatePicker({ initialDate, onSave, onCancel }) {

        const nextMonth = () => setViewDate(new Date(year, month + 1, 1));

        const handleSave = () => {
          let h = parseInt(hh, 10) || 0;
          let m = parseInt(mm, 10) || 0;
          if (h > 23) h = 23; if (m > 59) m = 59;
          
          const finalDate = new Date(selectedDate);
          finalDate.setHours(h, m, 0, 0);
          onSave(finalDate.toISOString());
        };

        const handleTimeChange = (type, val) => {
          const num = val.replace(/\D/g, '').slice(0, 2);
          if (type === 'hh') {
            if (parseInt(num) > 23) setHh('23'); else setHh(num);
          } else {
            if (parseInt(num) > 59) setMm('59'); else setMm(num);
          }
        };

          <div style={{ width: '100%' }}>
            <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: '16px' }}>

                    }}
                  >
                    {dayNum}
                    {isToday && !isSelected && <div style={{ width: 4, height: 4, borderRadius: 2, background: 'var(--primary)' }} />}

                <Check size={18} strokeWidth={2.5} /> Сохранить

            </div>

      }

'var(--primary)' : 'var(--text-muted)', border: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
                            <Hash size={20} />
                          </button>

                          {activeTab === 'post' && (
                            <>
                              {postCat !== 'lost_found' && postCat !== 'confessions' && (
                                <button onClick={() => setHasPoll(!hasPoll)} className="spring-btn" style={{ width: 40, height: 40, borderRadius: '20px', background: hasPoll ? 'rgba(212, 255, 0, 0.15)' : 'var(--surface-elevated)', color: hasPoll ? 'var(--primary)' : 'var(--text-muted)', border: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
                                  <BarChart2 size={20} />
                                </button>
                              )}
                              <button onClick={() => setIsAnonymous(!isAnonymous)} className="spring-btn" style={{ width: 40, height: 40, borderRadius: '20px', background: isAnonymous ? 'rgba(212, 255, 0, 0.15)' : 'var(--surface-elevated)', color: isAnonymous ? 'var(--primary)' : 'var(--text-muted)', border: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', opacity: postCat === 'confessions' ? 0.5 : 1, pointerEvents: postCat === 'confessions' ? 'none' : 'auto' }}>
                                <VenetianMask size={20} />
                              </button>
                            </>
                          )}

                          {activeTab === 'request' && (
                             <>
                               <button onClick={handleToggleReward} className="spring-btn" style={{ width: 40, height: 40, borderRadius: '20px', background: showReqReward || reqRewardType !== 'none' ? 'rgba(212, 255, 0, 0.15)' : 'var(--surface-elevated)', color: showReqReward || reqRewardType !== 'none' ? 'var(--primary)' : 'var(--text-muted)', border: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
                                 <Gift size={20} />
                               </button>
                               <button onClick={handleToggleDeadline} className="spring-btn" style={{ width: 40, height: 40, borderRadius: '20px', background: showReqDeadline || reqDeadlineType !== '24h' ? 'rgba(212, 255, 0, 0.15)' : 'var(--surface-elevated)', color: showReqDeadline || reqDeadlineType !== '24h' ? 'var(--primary)' : 'var(--text-muted)', border: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
                                 <Clock size={20} />
                               </button>
                             </>
                          )}
                        </div>

                        <button onClick={handleSend} className="spring-btn" style={{

                            background: canSend ? 'var(--primary)' : 'var(--surface-elevated)', 
                            color: canSend ? '#000' : 'var(--text-muted)', 
                            border: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', 
                            transition: 'all 0.3s'

                          <Send size={18} style={{ marginLeft: canSend ? '2px' : '0', transition: 'all 0.2s' }}/>
                        </button>
                      </div>
                  </div>

                </div>
              </div>
            </div>

            {/* ==========================================
                ГЛОБАЛЬНЫЙ КАЛЕНДАРЬ (СИСТЕМНАЯ КЛАВИАТУРА)
                Выезжает с самого низа, скрывая тулбар!
                ========================================== */}
            <div style={{
              position: 'absolute', inset: 0, zIndex: 3000,
              pointerEvents: (showEventPicker || showReqPicker) ? 'auto' : 'none',
              display: 'flex', flexDirection: 'column', justifyContent: 'flex-end'
            }}>
              {/* Затемнение остального интерфейса */}
              <div 
                onClick={() => { setShowEventPicker(false); setShowReqPicker(false); }}
                style={{ position: 'absolute', inset: 0, background: 'rgba(0,0,0,0.7)', backdropFilter: 'blur(2px)', opacity: (showEventPicker || showReqPicker) ? 1 : 0, transition: 'opacity 0.3s' }} 
              />
              
              {/* Сама шторка календаря */}
              <div style={{
                background: 'var(--surface)', borderTop: '1px solid var(--border)',
                borderTopLeftRadius: '24px', borderTopRightRadius: '24px',
                padding: '24px 16px calc(env(safe-area-inset-bottom, 20px) + 16px)',
                transform: (showEventPicker || showReqPicker) ? 'translateY(0)' : 'translateY(100%)',
                transition: 'transform 0.4s cubic-bezier(0.32, 0.72, 0, 1)',
                position: 'relative'
              }}>
                {showEventPicker && (
                  <SmartDatePicker 
                    initialDate={eventCustomDate} 
                    onSave={(iso) => { setEventCustomDate(iso); setShowEventPicker(false); }} 
                    onCancel={() => { setShowEventPicker(false); if(!eventCustomDate) setEventDate('today'); }} 
                  />
                )}
                {showReqPicker && (
                  <SmartDatePicker 
                    initialDate={reqCustomDate} 
                    onSave={(iso) => { setReqCustomDate(iso); setShowReqPicker(false); }} 
                    onCancel={() => { setShowReqPicker(false); if(!reqCustomDate) setReqDeadlineType('24h'); }} 
                  />
                )}
              </div>
            </div>

          </div>
        );
      }
