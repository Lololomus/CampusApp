                <h3 style={{ margin: '0 0 16px 0', fontSize: '14px', textTransform: 'uppercase', color: 'var(--text-muted)', fontWeight: 700 }}>Цели и интересы</h3>
                
                <div style={{ display: 'flex', flexWrap: 'wrap', gap: '8px' }}>
                  {profile.goals.map(g => <div key={g} className="chip active">{g}</div>)}
                  {profile.interests.map(i => <div key={i} className="chip inactive">{i}</div>)}
                  {profile.goals.length === 0 && profile.interests.length === 0 && (
                    <div style={{ color: 'var(--text-muted)', fontSize: '15px' }}>Выбрать теги...</div>
                  )}
                </div>
              </div>

            </div>

            {/* Sticky Preview Button */}
            <div style={{ position: 'absolute', bottom: 0, left: 0, right: 0, padding: '16px', paddingBottom: 'calc(var(--safe-area-bottom) + 16px)', background: 'linear-gradient(to top, var(--bg) 60%, transparent)', pointerEvents: 'none' }}>
               <button className="spring-btn" style={{ width: '100%', padding: '16px', background: 'var(--surface-elevated)', borderRadius: '20px', border: '1px solid var(--border)', color: '#FFF', fontWeight: 700, fontSize: '16px', display: 'flex', alignItems: 'center', justifyContent: 'center', gap: '8px', pointerEvents: 'auto' }}>
                  <MessageCircle size={20} />
                  Как видят другие
               </button>
            </div>


            {/* === BOTTOM SHEETS === */}

            {/* BIO SHEET */}
            <BottomSheet isOpen={activeSheet === 'bio'} onClose={closeSheet} title="Обо мне">
              <textarea 
                className="input-standard"
                placeholder="Расскажи, кого ищешь и чем увлекаешься..."
                value={tempBio}
                onChange={e => setTempBio(e.target.value)}
                autoFocus
              />
              <button className="spring-btn" onClick={saveSheet} style={{ width: '100%', padding: '16px', background: 'var(--primary)', borderRadius: '16px', border: 'none', color: '#000', fontWeight: 700, fontSize: '16px', marginTop: '24px' }}>
                Сохранить
              </button>
            </BottomSheet>

            {/* ICEBREAKER SHEET (Двухшаговый флоу с розовым акцентом) */}
            <BottomSheet isOpen={activeSheet === 'icebreaker'} onClose={closeSheet} title={ibStep === 'list' ? 'Выбери ледокол' : 'Твой ледокол'}>
              
              {/* Шаг 1: Выбор вопроса из вертикального списка (Сгруппировано по темам) */}
              {ibStep === 'list' && (
                <div style={{ display: 'flex', flexDirection: 'column', gap: '24px' }}>
                  
                  {/* Кнопка кастомного вопроса теперь сверху для быстрого доступа */}
                  <button 
                    className="spring-btn prompt-list-item custom"
                    onClick={() => { 
                      triggerHaptic('light'); 
                      setTempIcebreaker({ prompt: '', answer: '', isCustom: true }); 
                      setIbStep('custom'); 
                    }}
                  >
                    <Edit3 size={18} color="var(--dating)" style={{ flexShrink: 0 }} />
                    <span>Написать свой вопрос...</span>
                  </button>

                  {ICEBREAKERS_CATEGORIZED.map((cat, catIdx) => (
                    <div key={catIdx}>
                      <div style={{ fontSize: '13px', color: 'var(--text-muted)', fontWeight: 700, marginBottom: '10px', textTransform: 'uppercase', letterSpacing: '0.5px', paddingLeft: '4px' }}>
                        {cat.title}
                      </div>
                      <div style={{ display: 'flex', flexDirection: 'column', gap: '8px' }}>
                        {cat.items.map((prompt, idx) => (
                          <button 
                            key={idx} 
                            className="spring-btn prompt-list-item"
                            onClick={() => { 
                              triggerHaptic('light'); 
                              setTempIcebreaker({ prompt, answer: '', isCustom: false }); 
                              setIbStep('answer'); 
                            }}
                          >
                            <span style={{ flex: 1 }}>{prompt}</span>
                            <ChevronRight size={18} color="rgba(255, 255, 255, 0.2)" style={{ flexShrink: 0 }} />
                          </button>
                        ))}
                      </div>
                    </div>
                  ))}
                </div>
              )}

              {/* Шаг 2: Ввод ответа для ГОТОВОГО вопроса */}
              {ibStep === 'answer' && (
                <div style={{ display: 'flex', flexDirection: 'column', height: '100%' }}>
                  <button className="spring-btn" onClick={() => { triggerHaptic('light'); setIbStep('list'); }} style={{ background: 'transparent', border: 'none', color: 'var(--text-muted)', display: 'flex', alignItems: 'center', gap: '4px', padding: '0 0 16px 0', fontSize: '15px', fontWeight: 600 }}>
                    <ChevronLeft size={18} /> Другой вопрос
                  </button>
                  
                  <div style={{ fontSize: '15px', color: 'var(--text-muted)', marginBottom: '12px' }}>{tempIcebreaker.prompt}</div>
                  
                  <textarea 
                    className="textarea-hero dating-accent"
                    placeholder="Твой ответ..."
                    value={tempIcebreaker.answer}
                    onChange={e => setTempIcebreaker({ ...tempIcebreaker, answer: e.target.value })}
                    rows={4}
                    autoFocus
                  />
                </div>
              )}

              {/* Шаг 2 (Альтернативный): КАСТОМНЫЙ вопрос и ответ */}
              {ibStep === 'custom' && (
                <div style={{ display: 'flex', flexDirection: 'column', height: '100%' }}>
                  <button className="spring-btn" onClick={() => { triggerHaptic('light'); setIbStep('list'); }} style={{ background: 'transparent', border: 'none', color: 'var(--text-muted)', display: 'flex', alignItems: 'center', gap: '4px', padding: '0 0 16px 0', fontSize: '15px', fontWeight: 600 }}>
                    <ChevronLeft size={18} /> К списку
                  </button>
                  
                  <div style={{ fontSize: '12px', color: 'var(--text-muted)', marginBottom: '4px', fontWeight: 700, textTransform: 'uppercase', letterSpacing: '0.5px' }}>Твой вопрос:</div>
                  <input 
                    className="input-line dating-accent"
                    placeholder="Например: Любимый трек для пробежки?"
                    value={tempIcebreaker.prompt}
                    onChange={e => setTempIcebreaker({ ...tempIcebreaker, prompt: e.target.value })}
                    autoFocus
                  />

                  <div style={{ fontSize: '12px', color: 'var(--text-muted)', marginTop: '24px', marginBottom: '8px', fontWeight: 700, textTransform: 'uppercase', letterSpacing: '0.5px' }}>Твой ответ:</div>
                  <textarea 
                    className="textarea-hero dating-accent"
                    placeholder="Начни печатать..."
                    value={tempIcebreaker.answer}
                    onChange={e => setTempIcebreaker({ ...tempIcebreaker, answer: e.target.value })}
                    rows={3}
                  />
                </div>
              )}

              {/* Общая кнопка сохранения для ввода (не показывается в списке) */}
              {ibStep !== 'list' && (
                <button 
                  className="spring-btn" 
                  onClick={saveSheet} 
                  disabled={!tempIcebreaker.answer.trim() || !tempIcebreaker.prompt.trim()} 
                  style={{ 
                    width: '100%', padding: '16px', borderRadius: '16px', border: 'none', fontWeight: 700, fontSize: '16px', marginTop: '24px', transition: 'all 0.3s',
                    background: (tempIcebreaker.answer.trim() && tempIcebreaker.prompt.trim()) ? 'var(--dating)' : 'var(--surface-elevated)', 
                    color: (tempIcebreaker.answer.trim() && tempIcebreaker.prompt.trim()) ? '#FFF' : 'var(--text-muted)'
                  }}
                >
                  Установить ледокол
                </button>
              )}
            </BottomSheet>

            {/* TAGS SHEET */}
            <BottomSheet isOpen={activeSheet === 'tags'} onClose={closeSheet} title="Кого ищешь?">
              
              <div style={{ fontSize: '13px', color: 'var(--text-muted)', fontWeight: 700, marginBottom: '12px', textTransform: 'uppercase' }}>Твои цели (до 2)</div>
              <div style={{ display: 'flex', flexWrap: 'wrap', gap: '8px', marginBottom: '32px' }}>
                {ALL_GOALS.map(tag => {
                  const isActive = tempTags.goals.includes(tag);
                  return (
                    <div key={tag} className={`chip spring-btn ${isActive ? 'active' : 'inactive'}`} onClick={() => toggleTag('goals', tag)}>
                      {tag}
                    </div>
                  );
                })}
              </div>

              <div style={{ fontSize: '13px', color: 'var(--text-muted)', fontWeight: 700, marginBottom: '12px', textTransform: 'uppercase' }}>Интересы (до 5)</div>
              <div style={{ display: 'flex', flexWrap: 'wrap', gap: '8px', marginBottom: '24px' }}>
                {ALL_INTERESTS.map(tag => {
                  const isActive = tempTags.interests.includes(tag);
                  return (
                    <div key={tag} className={`chip spring-btn ${isActive ? 'active' : 'inactive'}`} onClick={() => toggleTag('interests', tag)}>
                      {tag}
                    </div>
                  );
                })}
              </div>

              <button className="spring-btn" onClick={saveSheet} style={{ width: '100%', padding: '16px', background: 'var(--primary)', borderRadius: '16px', border: 'none', color: '#000', fontWeight: 700, fontSize: '16px' }}>
                Сохранить теги
              </button>
            </BottomSheet>

            {/* SETTINGS SHEET (Анкета поиска) */}
            <BottomSheet isOpen={activeSheet === 'settings'} onClose={closeSheet} title="Настройки поиска">
              
              {/* Кого ищу */}
              <div style={{ marginBottom: '24px' }}>
                <div style={{ fontSize: '13px', color: 'var(--text-muted)', marginBottom: '12px', fontWeight: 700, textTransform: 'uppercase' }}>Кого я ищу</div>
                <div style={{ display: 'flex', gap: '8px' }}>
                  {['Парни', 'Девушки', 'Все'].map(opt => (
                    <button 
                      key={opt} 
                      className={`chip spring-btn ${tempSettings.lookingFor === opt ? 'dating-active' : 'inactive'}`} 
                      onClick={() => { triggerHaptic('light'); setTempSettings({...tempSettings, lookingFor: opt}); }}
                      style={{ flex: 1, justifyContent: 'center' }}
                    >
                      {opt}
                    </button>
                  ))}
                </div>
              </div>

              {/* Мой возраст (специфично для дейтинга) */}
              <div style={{ marginBottom: '32px' }}>
                <div style={{ fontSize: '13px', color: 'var(--text-muted)', marginBottom: '8px', fontWeight: 700, textTransform: 'uppercase' }}>Мой возраст</div>
                <input 
                  type="number" 
                  className="input-line dating-accent" 
                  value={tempSettings.age} 
                  onChange={e => setTempSettings({...tempSettings, age: e.target.value})} 
                  placeholder="Например: 20"
                />
              </div>

              {/* Скрыть анкету (Тумблер) */}
              <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between', marginBottom: '32px', padding: '16px', background: 'var(--surface-elevated)', borderRadius: '16px', border: '1px solid var(--border)' }}>
                <div>
                  <div style={{ fontWeight: 700, fontSize: '16px', color: '#FFF' }}>Скрыть анкету</div>
                  <div style={{ fontSize: '13px', color: 'var(--text-muted)', marginTop: '4px', lineHeight: 1.4 }}>Тебя не будет видно в ленте,<br/>но ты сможешь общаться с мэтчами</div>
                </div>
                <div 
                  className={`toggle-switch ${tempSettings.isHidden ? 'active' : ''}`} 
                  onClick={() => { triggerHaptic('light'); setTempSettings({...tempSettings, isHidden: !tempSettings.isHidden}); }} 
                />
              </div>

              <button className="spring-btn" onClick={saveSheet} style={{ width: '100%', padding: '16px', background: 'var(--dating)', borderRadius: '16px', border: 'none', color: '#FFF', fontWeight: 700, fontSize: '16px' }}>
                Сохранить настройки
              </button>
            </BottomSheet>

            {/* PHOTOS GRID SHEET (Максимум 3 фото) */}
            <BottomSheet isOpen={activeSheet === 'photos'} onClose={closeSheet} title="Мои фотографии">
               <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr 1fr', gap: '12px', marginBottom: '16px' }}>
                  {tempPhotos.map((p, i) => (
                     <div key={i} className="spring-btn" onClick={() => { if(!p) { triggerHaptic('light'); const newArr = [...tempPhotos]; newArr[i] = 'https://images.unsplash.com/photo-1522556189639-b150ed9c4330?auto=format&fit=crop&q=80&w=800'; setTempPhotos(newArr); } }} style={{ aspectRatio: '3/4', background: 'var(--surface-elevated)', borderRadius: '12px', position: 'relative', overflow: 'hidden', border: p ? 'none' : '1px dashed var(--border)', display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
                        {p ? (
                           <>
                             <img src={p} style={{ width: '100%', height: '100%', objectFit: 'cover' }} />
                             <button onClick={(e) => { e.stopPropagation(); triggerHaptic('light'); const newArr = [...tempPhotos]; newArr[i] = null; setTempPhotos(newArr); }} style={{ position: 'absolute', bottom: 6, right: 6, background: 'rgba(0,0,0,0.6)', backdropFilter: 'blur(4px)', borderRadius: '50%', width: '24px', height: '24px', display: 'flex', alignItems: 'center', justifyContent: 'center', border: 'none', color: '#FFF' }}><X size={14}/></button>
                             {i === 0 && <div style={{ position: 'absolute', top: 6, left: 6, background: 'var(--dating)', color: '#FFF', fontSize: '9px', fontWeight: 800, padding: '2px 6px', borderRadius: '6px' }}>MAIN</div>}
                           </>
                        ) : (
                           <Plus size={24} color="var(--text-muted)" />
                        )}
                     </div>
                  ))}
               </div>
               <p style={{ fontSize: '13px', color: 'var(--text-muted)', marginBottom: '24px', textAlign: 'center' }}>В дейтинге достаточно 3-х лучших фотографий.<br/>Нажми на пустой слот, чтобы добавить.</p>
               <button className="spring-btn" onClick={saveSheet} style={{ width: '100%', padding: '16px', background: 'var(--dating)', borderRadius: '16px', border: 'none', color: '#FFF', fontWeight: 700, fontSize: '16px' }}>
                 Сохранить фото
               </button>
            </BottomSheet>

          </div>
        );
      };

      const ALL_INTERESTS = ['📚 Учеба', '🎉 Тусовки', '☕ Кофе', '💻 IT', '🎨 Творчество', '🏃‍♂️ Спорт', '🎮 Игры', '🎵 Музыка', '🎬 Кино', '✈️ Путешествия'];

      const root = createRoot(document.getElementById('root'));
      root.render(<DatingProfile />);
