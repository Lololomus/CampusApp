      const Zoomable = ({ children }) => {
        const [scale, setScale] = useState(1);
        const startDist = useRef(null);

        const handleTouchStart = (e) => {
          if (e.touches.length === 2) {
            const dx = e.touches[0].clientX - e.touches[1].clientX;
            const dy = e.touches[0].clientY - e.touches[1].clientY;
            startDist.current = Math.hypot(dx, dy);
          }
        };

        const handleTouchMove = (e) => {
          if (e.touches.length === 2 && startDist.current) {
            const dx = e.touches[0].clientX - e.touches[1].clientX;
            const dy = e.touches[0].clientY - e.touches[1].clientY;
            const currentDist = Math.hypot(dx, dy);
            // Ограничиваем зум от 1x до 4x
            const newScale = Math.min(Math.max(1, currentDist / startDist.current), 4);
            setScale(newScale);
          }
        };

        const handleTouchEnd = () => {
          setScale(1); // Возвращаем масштаб обратно при отпускании (как в Instagram)
          startDist.current = null;
        };

        return (
          <div
            onTouchStart={handleTouchStart}
            onTouchMove={handleTouchMove}
            onTouchEnd={handleTouchEnd}
            style={{
              display: 'inline-flex', alignItems: 'center', justifyContent: 'center',
              transition: scale === 1 ? 'transform 0.3s cubic-bezier(0.32, 0.72, 0, 1)' : 'none',
              transform: `scale(${scale})`,
              transformOrigin: 'center center',
              zIndex: scale > 1 ? 50 : 1,
              maxWidth: '100%', maxHeight: '100%'
            }}

            {children}
