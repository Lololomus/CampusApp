      const triggerNeonSparks = (rect) => {

        const y = rect.top + rect.height / 2;

        // Эмуляция Haptic Feedback (на реальном девайсе использовать Telegram API)
        if (window.navigator && window.navigator.vibrate) window.navigator.vibrate([30, 50, 30]);

        for (let i = 0; i < 40; i++) {
          const spark = document.createElement('div');
          spark.style.position = 'fixed';
          spark.style.left = x + 'px';
          spark.style.top = y + 'px';
          // Микс из точек и коротких линий
          const isLine = Math.random() > 0.5;
          spark.style.width = isLine ? (8 + Math.random() * 16) + 'px' : (4 + Math.random() * 5) + 'px';
          spark.style.height = isLine ? '2px' : spark.style.width;
          spark.style.backgroundColor = Math.random() > 0.3 ? 'var(--primary)' : '#FFF';
          spark.style.borderRadius = isLine ? '2px' : '50%';
          spark.style.pointerEvents = 'none';
          spark.style.zIndex = '9999';
          document.body.appendChild(spark);

          const angle = Math.random() * Math.PI * 2;
          const velocity = 80 + Math.random() * 140;
          const tx = Math.cos(angle) * velocity;
          const ty = Math.sin(angle) * velocity;
          const rot = Math.random() * 360;

          const animation = spark.animate([
            { transform: `translate(-50%, -50%) rotate(${rot}deg) scale(1)`, opacity: 1 },
            { transform: `translate(calc(-50% + ${tx}px), calc(-50% + ${ty}px)) rotate(${rot + 90}deg) scale(0)`, opacity: 0 }
          ], {
            duration: 500 + Math.random() * 500,
            easing: 'cubic-bezier(0.25, 1, 0.5, 1)',
            fill: 'forwards'
          });

          animation.onfinish = () => spark.remove();
        }

                          <span className="uni-city">{u.full} • {u.city}</span>
                        </div>
                      </div>
                    ))}
                  </div>
                </div>
              )}

              {(selectedUni || isCustomMode) && (
                <div className="animate-scale" style={{ display: 'flex', flexDirection: 'column' }}>
                  
                  {selectedUni && (
                    <div style={{ background: 'var(--surface)', border: '1px solid var(--border)', borderRadius: '24px', padding: '16px', marginBottom: '32px', display: 'flex', alignItems: 'center', justifyContent: 'space-between' }}>
                      <div style={{ display: 'flex', alignItems: 'center', gap: '16px' }}>
                        <div className="uni-avatar" style={{ background: selectedUni.color, width: 64, height: 64, borderRadius: '18px', fontSize: '20px', overflow: 'hidden' }}>
                          {selectedUni.logo ? (
                            <img src={selectedUni.logo} alt={selectedUni.name} style={{ width: '100%', height: '100%', objectFit: 'cover' }} />
                          ) : (
                            selectedUni.name.charAt(0)
                          )}
                        </div>
                        <div className="uni-info">
                          <span className="uni-name" style={{ fontSize: '20px' }}>{selectedUni.name}</span>
                          <span className="uni-city">{selectedUni.city}</span>
                        </div>
                      </div>
                      <button className="spring-btn" onClick={handleResetUni} style={{ background: 'var(--surface-elevated)', border: 'none', color: 'var(--text)', padding: '8px 16px', borderRadius: '12px', fontSize: '14px', fontWeight: 600 }}>

                      <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: '16px' }}>
                        <span style={{ fontSize: '16px', fontWeight: 700, color: '#FFF' }}>Свой ВУЗ</span>

                          К списку

                        autoFocus

                        </div>
                      </div>
                    </div>

                    <h3 style={{ margin: '0 0 16px', fontSize: '18px', fontWeight: 700, color: '#FFF' }}>Какой курс?</h3>

              {!selectedUni && !isCustomMode && (
                <div className="spring-btn" onClick={() => setIsCustomMode(true)} style={{ display: 'flex', alignItems: 'center', gap: '12px', padding: '12px 16px', background: 'var(--surface)', borderRadius: '16px', border: '1px solid var(--border)', marginBottom: '16px' }}>

                    <Plus size={18} color="var(--text-muted)" />
                  </div>
                  <div style={{ display: 'flex', flexDirection: 'column' }}>
                    <span style={{ fontSize: '15px', fontWeight: 600, color: '#FFF' }}>Нет моего ВУЗа</span>
                    <span style={{ fontSize: '13px', color: 'var(--text-muted)' }}>Ввести вручную (через модерацию)</span>
                  </div>
                </div>
              )}

              <button 
                className={`spring-btn primary-cta ${finishStatus === 'success' ? 'btn-success-state' : ''}`} 
                onClick={handleFinishClick} 
                disabled={!isValid || finishStatus !== 'idle'}

              >
                {finishStatus === 'idle' && <><Check size={20} strokeWidth={3} /> Завершить</>}

                {finishStatus === 'success' && <><Zap size={20} fill="#000" /> Готово!</>}
              </button>
            </div>
          </div>
        );
      };

      function App() {
        const [authOpen, setAuthOpen] = useState(true);
        const [step, setStep] = useState(0); 
        const [userData, setUserData] = useState({});
        const [toast, setToast] = useState(null);

        const startRegistration = () => {
          setAuthOpen(false);
          setStep(1);
        };

        const goNext = () => setStep(2);
        const goBack = () => setStep(1);

        const showSuccessToast = () => {
          setToast('Профиль успешно создан! ⚡️');
          setTimeout(() => setToast(null), 3000);
          // В реальном приложении здесь был бы setStep(0) и редирект на ленту.
          // В моке мы оставляем пользователя на экране, чтобы он мог понажимать кнопку.
        };

        return (
          <div style={{ 
            position: 'relative', width: '100%', maxWidth: '430px', height: '100%', 
            backgroundColor: 'var(--bg)', overflow: 'hidden', display: 'flex', flexDirection: 'column',
            boxShadow: '0 0 40px rgba(0,0,0,0.5)'
          }}>
            
            {/* Глобальный премиум Toast */}
            {toast && (
              <div className="toast-enter" style={{
                position: 'absolute', top: 'max(44px, env(safe-area-inset-top))', left: '50%',
                background: 'var(--surface-elevated)', border: '1px solid var(--primary)',
                padding: '12px 20px', borderRadius: '24px', color: '#FFF', fontWeight: 600,
                zIndex: 9999, display: 'flex', alignItems: 'center', gap: '8px',
                boxShadow: '0 12px 32px rgba(212, 255, 0, 0.15)'
              }}>
                <Zap size={18} fill="var(--primary)" color="var(--primary)" />
                {toast}
              </div>
            )}

            <div style={{ flex: 1, padding: 'max(44px, env(safe-area-inset-top)) 20px 80px', display: 'flex', flexDirection: 'column', gap: '16px', filter: authOpen || step > 0 ? 'blur(12px) brightness(0.4)' : 'none', transition: 'filter 0.5s ease' }}>
              <div style={{ height: '140px', background: 'var(--surface)', borderRadius: '24px', border: '1px solid var(--border)' }} />
              <div style={{ height: '220px', background: 'var(--surface)', borderRadius: '24px', border: '1px solid var(--border)' }} />
              <div style={{ height: '120px', background: 'var(--surface)', borderRadius: '24px', border: '1px solid var(--border)' }} />
            </div>

            {authOpen && <AuthSheet onStart={startRegistration} onDismiss={() => setAuthOpen(false)} />}

            <div style={{ 
              position: 'absolute', inset: 0, zIndex: 2000, display: 'flex', width: '200%',
              transform: `translateX(${step === 1 ? '0' : step === 2 ? '-50%' : '100%'})`,
              transition: 'transform 0.5s cubic-bezier(0.32, 0.72, 0, 1), opacity 0.4s',
              opacity: step > 0 ? 1 : 0, pointerEvents: step > 0 ? 'auto' : 'none'
            }}>
              <div style={{ width: '50%', height: '100%', position: 'relative' }}>
                 <StepAboutYou data={userData} onUpdate={d => setUserData({...userData, ...d})} onNext={goNext} />
              </div>
              <div style={{ width: '50%', height: '100%', position: 'relative' }}>
                 <StepEducation data={userData} onUpdate={d => setUserData({...userData, ...d})} onBack={goBack} onFinish={showSuccessToast} />
              </div>
            </div>

            {!authOpen && step === 0 && (
              <div style={{ position: 'absolute', bottom: 40, width: '100%', display: 'flex', justifyContent: 'center' }}>
                <button className="spring-btn" onClick={() => setAuthOpen(true)} style={{ background: 'var(--surface-elevated)', border: '1px solid var(--border)', color: '#FFF', padding: '14px 28px', borderRadius: '20px', fontWeight: 700, fontSize: '15px' }}>
                  Открыть заново
                </button>
              </div>
            )}
          </div>
        );
      }

      const root = createRoot(document.getElementById('root'));
      root.render(
        <ErrorBoundary>
          <App />
        </ErrorBoundary>
      );
    </script>
