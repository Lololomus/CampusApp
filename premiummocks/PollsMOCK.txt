                {!showExplanationInput && !explanation ? (
                  <button 
                    onClick={() => setShowExplanationInput(true)}
                    className="spring-btn"
                    style={{ background: 'transparent', border: 'none', color: 'var(--text-muted)', fontSize: '13px', fontWeight: 600, display: 'flex', alignItems: 'center', gap: '6px', padding: '8px 0', cursor: 'pointer' }}
                  >
                    <Plus size={14} strokeWidth={2.5} /> Добавить объяснение
                  </button>
                ) : (
                  <div className="poll-option-animate" style={{ display: 'flex', gap: '12px' }}>
                    <div style={{ flex: 1, background: 'rgba(212, 255, 0, 0.03)', border: '1px solid rgba(212, 255, 0, 0.15)', borderRadius: '12px', padding: '12px 14px', display: 'flex', alignItems: 'flex-start', gap: '8px' }}>
                      <Lightbulb size={16} color="var(--primary)" style={{ flexShrink: 0, marginTop: '2px' }} />

      // ==========================================
      // 2. УНИВЕРСАЛЬНЫЙ VIEWER (Опрос и Викторина в одном стиле)
      // ==========================================
      const PollViewer = ({ poll }) => {
        const [hasVoted, setHasVoted] = useState(false);
        const [selectedOption, setSelectedOption] = useState(null);

        const isQuiz = poll.type === 'quiz';
        
        // Фейковые данные для демо
        const totalVotes = hasVoted ? 325 : 324;
        const mockPercentages = isQuiz 
          ? poll.options.map((_, i) => i === poll.correctOption ? 75 : 25 / Math.max(1, poll.options.length - 1))
          : poll.options.map((_, i) => i === 0 ? 60 : 40 / Math.max(1, poll.options.length - 1));

        const handleVote = (index) => {
          if (hasVoted) return;

          // В викторине клик по варианту сразу отправляет голос
          if (isQuiz) {
            setSelectedOption(index);
            setHasVoted(true);
            if (index === poll.correctOption) {
              confetti({ particleCount: 80, spread: 60, origin: { y: 0.6 }, colors: ['#D4FF00', '#32D74B', '#FFFFFF'] });
            }
          } else {
            // В обычном опросе юзер просто выбирает вариант (до нажатия "Голосовать")
            setSelectedOption(index);
          }
        };

        const submitRegularVote = () => {
          if (selectedOption !== null) {
            setHasVoted(true);
          }
        }

        return (
          <div style={{ background: 'var(--bg)', borderRadius: '16px', padding: '16px', width: '100%', border: '1px solid var(--border)' }}>
            
            {/* ШАПКА: БЕЙДЖ И ВОПРОС */}
            <div style={{ display: 'flex', alignItems: 'center', gap: '8px', marginBottom: '12px' }}>
              <span style={{ 
                background: isQuiz ? 'rgba(10, 132, 255, 0.15)' : 'rgba(212, 255, 0, 0.15)', 
                color: isQuiz ? 'var(--info)' : 'var(--primary)', 
                padding: '4px 8px', borderRadius: '6px', fontSize: '10px', fontWeight: 800, letterSpacing: '0.5px' 
              }}>
                {isQuiz ? 'ВИКТОРИНА' : 'ОПРОС'}
              </span>
              {poll.isAnonymous && <span style={{ fontSize: '12px', color: 'var(--text-muted)' }}>Анонимно</span>}
            </div>
            
            <h3 style={{ margin: '0 0 16px', fontSize: '16px', fontWeight: 700, color: '#FFF' }}>{poll.question}</h3>
            
            {/* СПИСОК ВАРИАНТОВ */}
            <div style={{ display: 'flex', flexDirection: 'column', gap: '8px' }}>
              {poll.options.map((opt, i) => {
                const isSelected = selectedOption === i;
                const isCorrect = isQuiz && i === poll.correctOption;
                
                // === ЛОГИКА ОТОБРАЖЕНИЯ ПОСЛЕ ГОЛОСОВАНИЯ ===
                if (hasVoted) {
                  let bgColor = 'var(--surface-input)';
                  let borderColor = 'transparent';
                  let fillBg = 'rgba(255,255,255,0.08)';
                  let textColor = '#FFF';
                  let Icon = null;

                  if (isQuiz) {
                    if (isCorrect) {
                      borderColor = 'var(--success)';
                      fillBg = 'var(--success-bg)';
                      textColor = 'var(--success)';
                      Icon = <CheckCircle2 size={18} color="var(--success)" />;
                    } else if (isSelected && !isCorrect) {
                      borderColor = 'var(--error)';
                      fillBg = 'var(--error-bg)';
                      textColor = 'var(--error)';
                      Icon = <XCircle size={18} color="var(--error)" />;
                    }
                  } else {
                    // Обычный опрос (результаты)
                    if (isSelected) {
                      borderColor = 'var(--primary)';
                      fillBg = 'rgba(212, 255, 0, 0.15)';
                      textColor = 'var(--primary)';
                      Icon = <CheckCircle2 size={18} color="var(--primary)" />;
                    }
                  }

                  return (
                    <div key={i} style={{ position: 'relative', height: '48px', borderRadius: '10px', background: bgColor, border: `1px solid ${borderColor}`, overflow: 'hidden', display: 'flex', alignItems: 'center', padding: '0 12px', transition: 'all 0.3s ease' }}>
                      <div className="poll-bar-fill" style={{ position: 'absolute', top: 0, left: 0, bottom: 0, background: fillBg, '--target-width': `${mockPercentages[i]}%` }} />
                      <div style={{ position: 'relative', zIndex: 1, display: 'flex', justifyContent: 'space-between', width: '100%', alignItems: 'center' }}>
                        <div style={{ display: 'flex', alignItems: 'center', gap: '10px' }}>
                          {Icon}
                          <span style={{ fontSize: '14px', fontWeight: (isSelected || isCorrect) ? 600 : 500, color: textColor }}>{opt}</span>
                        </div>
                        <span style={{ fontSize: '13px', fontWeight: 600, color: textColor }}>{Math.round(mockPercentages[i])}%</span>
                      </div>
                    </div>
                  );
                }

                // === ЛОГИКА ОТОБРАЖЕНИЯ ДО ГОЛОСОВАНИЯ ===
                let preVoteBg = 'var(--surface-input)';
                let preVoteBorder = 'transparent';
                
                if (!isQuiz && isSelected) {
                  preVoteBg = 'rgba(212, 255, 0, 0.05)';
                  preVoteBorder = 'var(--primary)';
                }

                return (
                  <div key={i} onClick={() => handleVote(i)} className="spring-btn" style={{ display: 'flex', alignItems: 'center', gap: '12px', padding: '14px 12px', borderRadius: '10px', background: preVoteBg, border: `1px solid ${preVoteBorder}`, cursor: 'pointer', transition: 'all 0.2s ease' }}>
                    {!isQuiz && (
                       <div style={{ width: 20, height: 20, borderRadius: '50%', border: `2px solid ${isSelected ? 'var(--primary)' : 'var(--text-muted)'}`, display: 'flex', alignItems: 'center', justifyContent: 'center', background: isSelected ? 'var(--primary)' : 'transparent' }}>
                         {isSelected && <Check size={12} color="#000" strokeWidth={3} />}
                       </div>
                    )}
                    <span style={{ fontSize: '14px', fontWeight: 500, color: '#FFF' }}>{opt}</span>
                  </div>
                );
              })}
            </div>

            {/* ОБЪЯСНЕНИЕ ВИКТОРИНЫ (После ответа) */}
            {hasVoted && isQuiz && poll.explanation && (
              <div className="poll-option-animate" style={{ marginTop: '16px', paddingLeft: '12px', borderLeft: `2px solid ${selectedOption === poll.correctOption ? 'var(--success)' : 'var(--error)'}` }}>
                 <div style={{ fontSize: '13px', fontWeight: 700, color: selectedOption === poll.correctOption ? 'var(--success)' : 'var(--error)', marginBottom: '4px' }}>
                   {selectedOption === poll.correctOption ? '🎉 Правильно!' : '💡 Ошибочка!'}
                 </div>
                 <div style={{ fontSize: '13px', color: '#EAEAEA', lineHeight: 1.4, whiteSpace: 'pre-wrap' }}>
                   {poll.explanation}
                 </div>
              </div>
            )}

            {/* ФУТЕР С ИНФОЙ И КНОПКОЙ */}
            <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginTop: '16px' }}>
              <span style={{ fontSize: '12px', color: 'var(--text-muted)' }}>{totalVotes} голосов</span>
              
              {/* Кнопка "Голосовать" (Только для обычного опроса, если еще не голосовали) */}
              {!hasVoted && !isQuiz && (
                <button 
                  onClick={submitRegularVote} 
                  className="spring-btn" 
                  style={{ 
                    background: selectedOption !== null ? 'var(--primary)' : 'rgba(255,255,255,0.1)', 
                    color: selectedOption !== null ? '#000' : 'var(--text-muted)', 
                    border: 'none', padding: '8px 20px', borderRadius: '12px', fontSize: '13px', fontWeight: 700, 
                    pointerEvents: selectedOption !== null ? 'auto' : 'none' 
                  }}
                >
                  Голосовать
                </button>
              )}
            </div>

          </div>
        );
      };
