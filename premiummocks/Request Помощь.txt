                      <div style={{ position: 'absolute', bottom: 0, left: 0, right: 0, height: '48px', background: `linear-gradient(to bottom, transparent, ${theme.surface})`, pointerEvents: 'none', opacity: !isTextExpanded || (isTextExpanded && !scrollState.isBottom) ? 1 : 0, transition: 'opacity 0.3s', zIndex: 1 }} />
                    )}

                  </div>
                  {req.text.length > 150 && !isTextExpanded && (
                    <button 
                      className="spring-btn"
                      onClick={() => setIsTextExpanded(true)}
                      style={{ background: 'none', border: 'none', color: theme.primary, fontSize: '14px', fontWeight: 600, padding: '8px 0 0', cursor: 'pointer' }}
                    >
                      Показать ещё
                    </button>
                  )}
                </div>

                {/* Прогресс-бар FOMO в модалке */}
                {req.targetHelpers > 0 && !isInProgress && (
                  <div style={{ marginBottom: '24px', background: 'rgba(255,255,255,0.03)', borderRadius: '16px', padding: '16px', border: `1px solid ${theme.border}` }}>
                    <div style={{ display: 'flex', justifyContent: 'space-between', fontSize: '14px', fontWeight: 600, marginBottom: '10px' }}>
                      <span style={{ color: '#FFF' }}>Собрано: <span style={{ color: theme.primary }}>{req.currentHelpers} человека</span></span>
                      <span style={{ color: theme.textMuted }}>Цель: {req.targetHelpers}</span>
                    </div>
                    <div style={{ height: '8px', background: 'rgba(255,255,255,0.1)', borderRadius: '4px', overflow: 'hidden' }}>
                      <div style={{ height: '100%', width: `${(req.currentHelpers / req.targetHelpers) * 100}%`, background: theme.primary, borderRadius: '4px', transition: 'width 0.3s ease' }} />
                    </div>
                  </div>
                )}

                {/* Мета данные дат */}
                <div style={{ background: 'rgba(255,255,255,0.03)', border: `1px solid ${theme.border}`, borderRadius: '16px', padding: '16px', display: 'flex', flexDirection: 'column', gap: '12px' }}>
                  <div style={{ display: 'flex', alignItems: 'center', gap: '10px', color: theme.textMuted, fontSize: '13px', fontWeight: 500 }}>
                    <Calendar size={16} /> Создано: {req.createdAt}
                  </div>
                  <div style={{ display: 'flex', alignItems: 'center', gap: '10px', color: theme.textMuted, fontSize: '13px', fontWeight: 500 }}>
                    <Clock size={16} /> Актуально до: <span style={{ color: isInProgress ? theme.textMuted : (req.isUrgent ? theme.error : '#FFF'), fontWeight: isInProgress ? 500 : (req.isUrgent ? 700 : 500) }}>{req.deadlineAt}</span>
                  </div>
                </div>

              </div>

              {/* Фиксированный Sticky Footer для кнопок */}
              <div style={{ 
                padding: '16px 20px', 
                paddingBottom: 'calc(16px + env(safe-area-inset-bottom, 20px))',
                background: theme.surface, 
                borderTop: `1px solid ${theme.border}`,
                display: 'flex', gap: '12px'
              }}>
                {req.isMine ? (
                  <button onClick={onClose} className="spring-btn" style={{ flex: 1, padding: '16px', borderRadius: '16px', background: theme.surfaceElevated, border: 'none', color: '#FFF', fontSize: '16px', fontWeight: 700 }}>
                    Закрыть таску (Решено)
                  </button>
                ) : isInProgress ? (
                   // Заблокированная кнопка, если уже в работе
                   <button onClick={onClose} style={{ flex: 1, padding: '16px', borderRadius: '16px', background: theme.surfaceElevated, border: 'none', color: theme.textMuted, fontSize: '16px', fontWeight: 700, display: 'flex', alignItems: 'center', justifyContent: 'center', gap: '8px' }}>
                     <Lock size={20} /> Исполнитель уже найден
                   </button>
                ) : (
                  // Главная CTA кнопка
                  <button onClick={onClose} className="spring-btn" style={{ flex: 1, padding: '16px', borderRadius: '16px', background: theme.primary, border: 'none', color: '#000', fontSize: '16px', fontWeight: 800, display: 'flex', alignItems: 'center', justifyContent: 'center', gap: '8px', boxShadow: `0 8px 24px rgba(212, 255, 0, 0.2)` }}>
                    <MessageSquare size={20} strokeWidth={2.5} /> Откликнуться
                  </button>
                )}
              </div>
            </div>
          </div>
        );
      };
