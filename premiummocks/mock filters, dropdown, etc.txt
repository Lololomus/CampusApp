        border: 'rgba(255, 255, 255, 0.08)',

        return (

            position: 'absolute', bottom: '80px', left: '16px', right: '16px', 
            zIndex: 9999, display: 'flex', flexDirection: 'column',

          </div>

      };

      const ToastItem = ({ toast, onRemove }) => {
        const [progress, setProgress] = useState(100);
        const duration = 3000;

        useEffect(() => {
          const startTime = Date.now();
          const timer = setInterval(() => {
            const passed = Date.now() - startTime;
            const remaining = Math.max(0, 100 - (passed / duration) * 100);
            setProgress(remaining);
            if (passed >= duration) {
              clearInterval(timer);
              onRemove();
            }
          }, 16);
          return () => clearInterval(timer);
        }, []);

        const config = {
          success: { icon: CheckCircle2, color: theme.success, bg: 'rgba(50, 215, 75, 0.15)' },
          error: { icon: AlertTriangle, color: theme.error, bg: 'rgba(255, 69, 58, 0.15)' },
          info: { icon: Info, color: theme.info, bg: 'rgba(10, 132, 255, 0.15)' }

            background: theme.card, width: '100%', maxWidth: '340px',

          </div>
        );
      };

'#FFF' : '#000', fontSize: '14px', fontWeight: 600 }}>{confirmText}</button>

                        left: `${(priceRange[0] / MAX_PRICE) * 100}%`, right: `${100 - (priceRange[1] / MAX_PRICE) * 100}%`
